function updateTotals() {
    let commission = parseFloat(document.getElementById('commission').value) || 0;
    let recurring = parseFloat(document.getElementById('recurring').value) || 0;
    let subscriptions = parseInt(document.getElementById('subscriptions').value) || 0;
    let upfrontSales = parseInt(document.getElementById('upfrontSales').value) || 0;

    let monthlyTotal = recurring * subscriptions;
    let upfrontTotal = commission * upfrontSales;
    let totalEarned = monthlyTotal + upfrontTotal;

    document.getElementById('monthlyTotal').textContent = '$' + monthlyTotal.toFixed(2);
    document.getElementById('upfrontTotal').textContent = '$' + upfrontTotal.toFixed(2);
    document.getElementById('totalEarned').textContent = '$' + totalEarned.toFixed(2);
}

window.onload = function() {
    let today = new Date();
    let formatted = today.getFullYear()+'-'+(today.getMonth()+1).toString().padStart(2,'0')+'-'+today.getDate().toString().padStart(2,'0');
    document.getElementById('currentDate').value = formatted;
};

function printAgreement() {
    window.print();
}
